VAR a : INT;
@(NOP)
{(
    (a := 101);
    (PRINT a);
    (   
        (
            (
                @(NOP)
                (
                    ((a-105)?);
                    (a:=(a+1)); 
                    (PRINT a)
                )*
            );
            ((!(a-105))?)
        ) 
        U 
        (PRINT (a-100))
    ); 
    (a := 103);
    (PRINT a)
)}
@(NOP)
