VAR x : INT;
VAR y : INT;
VAR a : INT;
VAR b : INT;
VAR rem : INT;
@((x > 0) | (y > 0))
{(  
    (x := 234);
    (y := 16);
    (a := x);
    (b := y);
    (
        @(A dd ((dd < (a + b)) => ((PRED nod a b dd) => (PRED nod x y dd))) )
        WHILE
        b
        DO
        (
            (rem := a);
            (
                @(rem >= 0)
                WHILE
                (rem >= b)
                DO
                (rem := (rem - b))
                DONE
            );
            (a := b);
            (b := rem)
        )
        DONE
    );
    (PRINT a)
)}
@((PRED nod x y rem) & (b == 0))
